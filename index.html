// If you do not support CSS transition, use of frame-based animation ($.fn.animate)
	// IE9 support
	
	
	$('#box').on('click', function() {
		$(this).transition({
			'x': 200,
			'y': 300
		}, 500);
		
		return false;
	});
	
	$('body').on('click', function() {
		$('#box').transition({
			'x': 0,
			'y': 0
		}, 500);
	});
